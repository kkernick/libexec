53 template <typename C, typename T> void split(C& working, const std::string_view& str, const T& delim, const bool& escape = false) {
54
55 // Ignore our delim if its contained within quotes.
56 bool wrapped = false;
57 for (size_t r_bound = 0, l_bound = 0; r_bound <= str.length(); ++r_bound) {
58 if (!wrapped && (contains(str[r_bound], delim) || r_bound == str.length())) {
59 if (r_bound != l_bound) emplace(working, std::string_view(&str[l_bound], r_bound - l_bound));
60 l_bound = r_bound + 1;
61 }
62
63 // If we hit a quote, we ignore the delimeters until we reach the match.
64 // We also skip over the quote itself when emplacing.
65 else if (escape && str[r_bound] == '\'') {
66 if (r_bound == l_bound)
67 ++l_bound;
68 else if (contains(str[r_bound + 1], delim)) {
69 emplace(working, std::string_view(&str[l_bound], r_bound - l_bound));
70 l_bound = r_bound + 1;
71 }
72 wrapped ^= 1;
73 }
74 }
75 }

250 template <class R = void, class T = std::initializer_list<std::string_view>> R execute(
251 const T& cmd,
252 const std::function<R(const int&, const int&)>& parser = [](const int& fd, const int& pid) {close(fd);},
253 const options& opts = {}
254 ) {
255
256 // Open a pipe for communication from the child
257 int output[2];
258 if (pipe(output) == -1) throw std::runtime_error("Failed to setup pipe");
259
260 // Open a pipe for comunication to the child, if needed.
261 int input[2] = {-1, -1};
262 if (!opts.in.empty()) {
263 if (pipe(input) == -1) throw std::runtime_error("Failed to setup pipe");
264 }
265
266 auto pid = fork();
267 if (pid < 0) throw std::runtime_error("Failed to fork");
268
269 else if (pid == 0) {
270 // Zip the command, open our standard files to the pipe.
271 auto argv = zip(cmd, opts.verbose);
272
273 close(output[0]);
274 switch (opts.cap) {
275 case STDOUT: dup2(output[1], STDOUT_FILENO); break;
276 case STDERR: dup2(output[1], STDERR_FILENO); break;
277 case ALL: dup2(output[1], STDOUT_FILENO); dup2(output[1], STDERR_FILENO); break;
278 default: break;
279 }
280
281 if (!opts.in.empty()) {
282 close(input[1]);
283 dup2(input[0], STDIN_FILENO);
284 close(input[1]);
285 }
286
287 // Close the pipe and execute.
288 close(output[1]);
289 execvp(argv[0], const_cast<char* const*>(argv.data()));
290 }
291
292 // If there is STDIN, send it to the child.
293 if (!opts.in.empty()) {
294 close(input[0]);
295 write(input[1], opts.in.c_str(), opts.in.length());
296 close(input[1]);
297 }
298
299 // Close the pipe and parser the other side.
300 close(output[1]);
301 return parser(output[0], pid);
302 }
